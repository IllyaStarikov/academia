\subsubsection{Implicit Differentiation}
Assume $p$ and $q$ are integers with $q \neq 0$. Then 

\begin{equation}
    \frac{d}{dx}\left( x^{p/q} \right) = \frac{p}{q} x^{p/q - 1}
\end{equation}
