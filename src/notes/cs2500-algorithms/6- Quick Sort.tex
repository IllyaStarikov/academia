\section{Quick Sort}
\begin{enumerate}
\item wall, current, pivot
    \begin{itemize}
    \item Array $\leq$ pivot $\leq$ Array
    \end{itemize}
\end{enumerate}

\begin{verbatim}
let wall = |, pivot = []
| 2 9 5 6 10 0 7 [4]
  2 | 9 5 6 10 0 7
    2 0 | 5 6 10 9 [7]
      0 2 -> 5 -> 6 | 10 9
        5 6 | 10 9

Output: 0 2 5 6 7 9 10
\end{verbatim}
