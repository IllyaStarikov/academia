\subsection{Substitution Method}
$2^n-1$ \\
\textbf{Base case}: $2-1 = 1$ \\
\textbf{Inductive}:

\begin{eqnarray} \\
T(n) & = & 2^n - 1  \\
T(n + 1) & = & 2^{n + 1} - 1 \\
& = & 2 \ T(n) + 1 \\
& = & 2(2^n - 1) \\
& = & 2^{n+1} - 1
\end{eqnarray}
